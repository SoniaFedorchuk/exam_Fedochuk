#include <iostream>
#include <iomanip>
#include <cstring>
using namespace std;
void Var1(int height);
void Var2(int height);
void CheckVar(char* variant);
void CheckHeight(char* height);
int main()
{
	char height[10];
	char variant[10];
	cout << "Enter height and variant of build 0 - single, 1 - double\n";
	cin >> height;
	cin.getline(variant, 10);

	CheckHeight(height);
	while (strlen(variant) == 0)
	{
		cout << "Enter variant!\n";
		cin.getline(variant, 10);
	}
	CheckVar(variant);
	if (variant[0] == '0')
		Var1(atoi(height));
	else if (variant[0] == '1')
		Var2(atoi(height));
	system("pause");
	return 0;
}
void Var1(int height)
{
	for (int i = 1; i <= height; i++)
	{
		cout << setfill(' ') << setw(height - i + 1) << " ";
		cout << setfill('#') << setw((i)) << "#";
		cout << "\n";
	}
}
